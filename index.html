Works using classes changings and "max-height" property

Section 1
let tabs = document.querySelectorAll('.acc3__tab');
let sections = document.querySelectorAll('.acc3__section');
let contents = document.querySelectorAll('.acc3__content');
									
for (let i = 0; i < contents.length; i++) { 
	sections[i].addEventListener('click', function () { 
		let active=document.querySelector('.acc3__content--active'); 
		let open=document.querySelector('.acc3__tab--open');
		if (contents[i].classList.contains('acc3__content--active') 
		&& contents[i].style.maxHeight !=null) {
			contents[i].classList.remove('acc3__content--active');
			tabs[i].classList.remove('acc3__tab--open');
			contents[i].style.maxHeight=null; 
		} else { 
			contents[i].classList.add('acc3__content--active');
			tabs[i].classList.add('acc3__tab--open');
			contents[i].style.maxHeight=contents[i].scrollHeight + "px";
			if (active) {
				active.classList.remove('acc3__content--active');
				open.classList.remove('acc3__tab--open'); 
				maxHeight=null;
			} 
		} 
	}); 
}
